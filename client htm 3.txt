var resto = {};
var cart = [];
var isTyping = false;

// ===== INIT =====
window.onload = function() {
  var params = new URLSearchParams(window.location.search);
  var data = params.get('data');

  if(!data) { showError(); return; }

  try {
    // Safe decode for Unicode/French characters
    var decoded = atob(data).split('').map(function(c) {
      return '%' + ('00' + c.charCodeAt(0).toString(16)).slice(-2);
    }).join('');
    resto = JSON.parse(decodeURIComponent(decoded));
    if(resto.color) document.documentElement.style.setProperty('--gold', resto.color);

    var av = document.getElementById('c-avatar');
    av.textContent = resto.emoji || '🍽️';
    av.style.background = (resto.color||'#D4A843') + '22';
    av.style.color = resto.color || '#D4A843';
    document.getElementById('c-name').textContent = resto.name || 'Restaurant';
    document.getElementById('c-city').textContent = resto.city || 'Brazzaville';
    document.title = (resto.name||'Restaurant') + ' — Commander';

    document.getElementById('loading').style.display = 'none';
    document.getElementById('app').style.display = 'flex';

    initChat();
    renderMenu();
  } catch(e) {
    showError();
  }
};

function showError() {
  document.getElementById('loading').style.display = 'none';
  document.getElementById('error-screen').style.display = 'flex';
}

// ===== TABS =====
function switchTab(id) {
  document.querySelectorAll('.panel').forEach(function(p){p.classList.remove('active')});
  document.querySelectorAll('.tab').forEach(function(t){t.classList.remove('active')});
  document.getElementById('panel-'+id).classList.add('active');
  var map = {chat:'ctab-chat',menu:'ctab-menu',order:'ctab-order'};
  if(map[id]) document.getElementById(map[id]).classList.add('active');
  if(id==='order') renderOrder();
  if(id==='menu') renderMenu();
}

// ===== CHAT =====
function initChat() {
  addMsg('ai', 'Bienvenue chez ' + (resto.name||'nous') + ' ! Je suis votre serveur. Dites-moi ce qui vous ferait plaisir — je connais chaque plat du menu et je peux vous aider a commander.');
  renderQuick();
}

function renderQuick() {
  var qs = ['Qu est-ce que vous recommandez ?', 'Voir le menu', 'Plat du jour ?', 'Je suis vegetarien'];
  document.getElementById('quick-row').innerHTML = qs.map(function(q) {
    return '<button class="q-btn" onclick="sendQuick(this.textContent)">' + q + '</button>';
  }).join('');
}

function addMsg(role, text) {
  var msgs = document.getElementById('messages');
  var div = document.createElement('div');
  div.className = 'msg ' + role;
  var color = resto.color || '#D4A843';
  var av = role==='ai'
    ? '<div class="av ai" style="background:' + color + '22;color:' + color + '">' + (resto.emoji||'🍽️') + '</div>'
    : '<div class="av user">👤</div>';
  var bub = role==='ai'
    ? '<div class="bub ai">' + text.replace(/\n/g,'<br>') + '</div>'
    : '<div class="bub user" style="background:' + color + '">' + text + '</div>';
  div.innerHTML = role==='ai' ? av+bub : bub+av;
  msgs.appendChild(div);
  msgs.scrollTop = msgs.scrollHeight;
}

function showTyping() {
  var msgs = document.getElementById('messages');
  var d = document.createElement('div');
  d.className='msg ai'; d.id='typing';
  var color = resto.color||'#D4A843';
  d.innerHTML = '<div class="av ai" style="background:' + color + '22;color:' + color + '">' + (resto.emoji||'🍽️') + '</div><div class="bub ai"><div class="typing-dots"><span></span><span></span><span></span></div></div>';
  msgs.appendChild(d); msgs.scrollTop=msgs.scrollHeight;
}

function removeTyping() { var el=document.getElementById('typing'); if(el) el.remove(); }

async function sendMsg() {
  var inp = document.getElementById('chat-input');
  var text = inp.value.trim();
  if(!text||isTyping) return;
  inp.value = '';
  addMsg('user', text);
  await getReply(text);
}

function sendQuick(t) { document.getElementById('chat-input').value=t; sendMsg(); }

async function getReply(userMsg) {
  isTyping = true;
  document.getElementById('send-btn').disabled = true;
  showTyping();

  var menu = resto.menu || [];
  var menuCtx = menu.length > 0
    ? menu.map(function(i){ return '- ' + i.nameFr + ' (' + (i.category||'Menu') + '): ' + Number(i.price).toLocaleString() + ' FCFA' + (i.desc?' — '+i.desc:''); }).join('\n')
    : 'Menu en cours de configuration.';

  var menuIds = JSON.stringify(menu.map(function(i){ return {id:i.id, name:i.nameFr}; }));

  var sys = 'Tu es le serveur IA de "' + (resto.name||'ce restaurant') + '" a ' + (resto.city||'Brazzaville') + '.\n\n' +
    'Personnalite : classe et charisme. Chaleureux sans en faire trop. Professionnel comme un excellent serveur. Une phrase suffit quand c est suffisant.\n\n' +
    'IMPORTANT : si le client veut commander un plat, reponds normalement ET ajoute a la fin de ta reponse exactement cette balise : [ADD_TO_CART:ID_DU_PLAT] en remplacant ID_DU_PLAT par le vrai id du plat dans cette liste : ' + menuIds + '\n\n' +
    'Exemple : si le client dit "je veux le poulet" et que l id du poulet est "d123", reponds : "Excellent choix ! Le Poulet Moambe est notre specialite. [ADD_TO_CART:d123]"\n\n' +
    'Si le client veut plusieurs plats, ajoute plusieurs balises.\n\n' +
    'Menu complet :\n' + menuCtx;

  try {
    var res = await fetch('/api/chat', {
      method: 'POST',
      headers: {'Content-Type': 'application/json'},
      body: JSON.stringify({system: sys, messages: [{role:'user', content: userMsg}]})
    });
    var data = await res.json();
    removeTyping();

    var reply = data.content && data.content[0] ? data.content[0].text : 'Desole, une erreur est survenue.';

    // Parse ADD_TO_CART tags
    var addedItems = [];
    var cleanReply = reply.replace(/\[ADD_TO_CART:([^\]]+)\]/g, function(match, id) {
      var item = (resto.menu||[]).find(function(i){ return i.id === id; });
      if(item) {
        addToCart(id, true);
        addedItems.push(item.nameFr);
      }
      return '';
    }).trim();

    addMsg('ai', cleanReply);

    if(addedItems.length > 0) {
      showCartNotif(addedItems);
    }

  } catch(e) {
    removeTyping();
    addMsg('ai', 'Connexion impossible. Reessayez.');
  }
  isTyping = false;
  document.getElementById('send-btn').disabled = false;
}

function showCartNotif(items) {
  var notif = document.getElementById('cart-notif');
  var text = document.getElementById('cart-notif-text');
  text.textContent = items.join(', ') + ' ajoute' + (items.length > 1 ? 's' : '') + ' au panier';
  notif.classList.add('show');
  setTimeout(function(){ notif.classList.remove('show'); }, 4000);
}

// ===== MENU =====
function renderMenu() {
  var c = document.getElementById('menu-wrap');
  var menu = resto.menu || [];

  if(!menu.length) {
    c.innerHTML = '<div class="empty-state"><div class="big">🍽️</div><p>Menu en cours de preparation</p></div>';
    return;
  }

  var cats = {};
  menu.forEach(function(item) {
    var cat = item.category || 'Menu';
    if(!cats[cat]) cats[cat] = [];
    cats[cat].push(item);
  });

  var html = '';
  Object.keys(cats).forEach(function(cat) {
    html += '<div class="cat-title">🍽️ ' + cat + '</div><div class="menu-grid">';
    cats[cat].forEach(function(item) {
      var topHtml = item.photo
        ? '<img class="m-card-img" src="' + item.photo + '" alt="' + item.nameFr + '">'
        : '<div class="m-card-body"><div class="m-emoji">' + (item.emoji||'🍽️') + '</div></div>';
      var bodyClass = item.photo ? 'm-card-body' : '';
      html += '<div class="m-card">' +
        topHtml +
        '<div class="m-card-body">' +
        '<div class="m-name">' + item.nameFr + '</div>' +
        '<div class="m-price">' + Number(item.price).toLocaleString() + ' FCFA</div>' +
        (item.desc ? '<div class="m-desc">' + item.desc + '</div>' : '') +
        '</div>' +
        '<button class="m-add" onclick="addToCart(\'' + item.id + '\', false)">+ Ajouter</button>' +
        '</div>';
    });
    html += '</div>';
  });
  c.innerHTML = html;
}

// ===== CART =====
function addToCart(id, fromAI) {
  var item = (resto.menu||[]).find(function(i){ return i.id === id; });
  if(!item) return;
  var ex = cart.find(function(c){ return c.id === id; });
  if(ex) ex.qty++;
  else cart.push(Object.assign({}, item, {qty:1}));
  updateBadge();
  if(!fromAI) showToast('✓ Ajoute !');
}

function updateBadge() {
  var total = cart.reduce(function(s,i){ return s+i.qty; }, 0);
  var b = document.getElementById('cart-badge');
  b.textContent = total;
  b.style.display = total > 0 ? 'inline-flex' : 'none';
}

function renderOrder() {
  var c = document.getElementById('order-wrap');
  var footer = document.getElementById('order-footer');
  if(!cart.length) {
    c.innerHTML = '<div class="empty-state"><div class="big">🛒</div><p>Panier vide</p><p style="margin-top:6px;font-size:0.78rem">Ajoutez des plats depuis le menu ou demandez au serveur</p></div>';
    footer.style.display = 'none';
    return;
  }
  footer.style.display = 'block';
  c.innerHTML = cart.map(function(item) {
    var imgHtml = item.photo
      ? '<img src="' + item.photo + '" style="width:44px;height:44px;object-fit:cover;border-radius:7px;flex-shrink:0">'
      : '<div style="font-size:1.4rem;flex-shrink:0">' + (item.emoji||'🍽️') + '</div>';
    return '<div class="o-item">' + imgHtml +
      '<div class="o-info"><div class="o-name">' + item.nameFr + '</div><div class="o-price">' + (Number(item.price)*item.qty).toLocaleString() + ' FCFA</div></div>' +
      '<div class="qty-ctrl"><button class="qty-btn" onclick="changeQty(\'' + item.id + '\',-1)">−</button><span class="qty">' + item.qty + '</span><button class="qty-btn" onclick="changeQty(\'' + item.id + '\',1)">+</button></div></div>';
  }).join('');
  var total = cart.reduce(function(s,i){ return s+Number(i.price)*i.qty; }, 0);
  document.getElementById('total-val').textContent = total.toLocaleString() + ' FCFA';
}

function changeQty(id, d) {
  var i = cart.findIndex(function(c){ return c.id === id; });
  if(i===-1) return;
  cart[i].qty += d;
  if(cart[i].qty <= 0) cart.splice(i, 1);
  updateBadge(); renderOrder();
}

function confirmOrder() {
  var count = cart.reduce(function(s,i){ return s+i.qty; }, 0);
  var total = cart.reduce(function(s,i){ return s+Number(i.price)*i.qty; }, 0);
  cart = []; updateBadge(); renderOrder();
  addMsg('ai', 'Commande confirmee ! ' + count + ' article(s) pour ' + total.toLocaleString() + ' FCFA.\nTemps estime : 20-30 min. Merci et bon appetit !');
  switchTab('chat');
  showToast('Commande confirmee !');
}

function showToast(msg) {
  var t = document.getElementById('toast');
  t.textContent = msg; t.classList.add('show');
  setTimeout(function(){ t.classList.remove('show'); }, 2500);
}
