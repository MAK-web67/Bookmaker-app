function renderMenu() {

  const menu = config.restaurantMenu || [];

  if (menu.length === 0) {
    c.innerHTML = `

    return;
  }

  // Grouper par catégorie
  const categories = {};
  menu.forEach(item => {
    const cat = item.category || (isFr ? 'Menu' : 'Menu');
    if (!categories[cat]) categories[cat] = [];
    categories[cat].push(item);
  });

  let html = '';
  Object.entries(categories).forEach(([cat, items]) => {
    html += `<div class="cat-title">🍽️ ${cat}</div><div class="menu-grid">`;
    html += items.map(item => `
      <div class="m-card" onclick="addToCart('${item.id}')">

        <div class="m-name">${isFr ? item.nameFr : (item.nameEn || item.nameFr)}</div>
        <div class="m-price">${Number(item.price).toLocaleString()} FCFA</div>
        <div class="m-desc">${item.description || ''}</div>
      </div>`).join('');
    html += '</div>';
  });
  c.innerHTML = html;
}


// ===== CART =====
function addToCart(id) {
  const menu = config.restaurantMenu || [];
  const item = menu.find(i => i.id === id);
  if (!item) return;
  const ex = cart.find(c => c.id === id);
  if (ex) ex.qty++;
  else cart.push({
    id: item.id,
    e: item.emoji || '🍽️',
    fr: item.nameFr,
    en: item.nameEn || item.nameFr,
    p: Number(item.price),
    qty: 1
  });
  updateBadge();
  showToast(T[lang].added);
}

}

        <button class="qty-btn" onclick="changeQty(${item.id},-1)">−</button>
        <span class="qty">${item.qty}</span>
        <button class="qty-btn" onclick="changeQty(${item.id},1)">+</button>
      </div>
    </div>`).join('');
  const total = cart.reduce((s,i)=>s+i.p*i.qty,0);
  document.getElementById('total-val').textContent = total.toLocaleString()+' FCFA';
  document.getElementById('confirm-btn').textContent = T[lang].confirmBtn;
}

function changeQty(id, d) {
  const i = cart.findIndex(c=>c.id===id);
  if(i===-1) return;
  cart[i].qty += d;
  if(cart[i].qty<=0) cart.splice(i,1);
  updateBadge(); renderOrder();
}

function confirmOrder() {
  if (!cart.length) return;
  const count = cart.reduce((s,i)=>s+i.qty,0);
  cart=[]; updateBadge(); renderOrder();
  showToast(T[lang].confirmed);
  addMsg('ai', T[lang].upgradeMsg(count));
  showTab('chat');
}

function addMenuItem() {
  const nameFr = document.getElementById('dish-nameFr') ? document.getElementById('dish-nameFr').value.trim() : '';
  const nameEn = document.getElementById('dish-nameEn') ? document.getElementById('dish-nameEn').value.trim() : '';
  const price = document.getElementById('dish-price') ? document.getElementById('dish-price').value : '';
  const emoji = document.getElementById('dish-emoji') ? (document.getElementById('dish-emoji').value.trim() || '🍽️') : '🍽️';
  const category = document.getElementById('dish-category') ? (document.getElementById('dish-category').value.trim() || (lang==='fr'?'Menu':'Menu')) : 'Menu';
  const description = document.getElementById('dish-desc') ? document.getElementById('dish-desc').value.trim() : '';

  if (!nameFr || !price) {
    showToast(lang==='fr' ? '⚠ Nom FR et prix requis' : '⚠ FR name and price required');
    return;
  }

  if (!config.restaurantMenu) config.restaurantMenu = [];
  config.restaurantMenu.push({
    id: 'dish_' + Date.now(),
    nameFr: nameFr,
    nameEn: nameEn || nameFr,
    price: Number(price),
    emoji: emoji,
    category: category,
    description: description
  });

  showToast(lang==='fr' ? '✓ Plat ajouté !' : '✓ Dish added!');
  renderSettings();
  renderMenu();
}

function deleteMenuItem(index) {
  if (!config.restaurantMenu) return;
  config.restaurantMenu.splice(index, 1);
  showToast(lang==='fr' ? '✓ Plat supprimé' : '✓ Dish removed');
  renderSettings();
  renderMenu();
}

function saveSettings() {
  const nameEl = document.getElementById('s-name');
  const descEl = document.getElementById('s-desc');
  const cityEl = document.getElementById('s-city');
  const typeEl = document.getElementById('s-type');
  if (nameEl) config.name = nameEl.value || config.name;
  if (descEl) config.desc = descEl.value || config.desc;
  if (cityEl) config.city = cityEl.value || config.city;
  if (typeEl) config.type = typeEl.value || config.type;
  applyConfig();
  showToast(lang==='fr'?'✓ Sauvegardé !':'✓ Saved!');
}
