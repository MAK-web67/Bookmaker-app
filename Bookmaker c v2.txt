async function getReply(userMsg) {
  isTyping=true;
  document.getElementById('send-btn').disabled=true;
  showTyping();

  var menuCtx = cfg.menu.length>0
    ? cfg.menu.map(function(i){ return '- ' + i.nameFr + ': ' + Number(i.price).toLocaleString() + ' FCFA'; }).join('\n')
    : 'Menu en cours de configuration.';

  var sys = cfg.lang==='fr'
    ? 'Tu es l assistante de "' + cfg.name + '", un ' + cfg.type + ' a ' + cfg.city + '. Professionnelle, efficace et agreable. Chaleureuse sans en faire trop. 2-3 phrases max. Menu:\n' + menuCtx
    : 'You are the assistant of "' + cfg.name + '", a ' + cfg.type + ' in ' + cfg.city + '. Professional, efficient, pleasant. Warm but not over the top. 2-3 sentences max. Menu:\n' + menuCtx;

  try {
    var res = await fetch('/api/chat', {
      method:'POST',
      headers:{'Content-Type':'application/json'},
      body:JSON.stringify({system:sys,messages:[{role:'user',content:userMsg}]})
    });
    var data = await res.json();
    removeTyping();
    addMsg('ai', data.content&&data.content[0] ? data.content[0].text : 'Desole, une erreur est survenue.');
  } catch(e) {
    removeTyping();
    addMsg('ai', cfg.lang==='fr' ? 'Connexion impossible. Reessayez.' : 'Connection failed. Please retry.');
  }
  isTyping=false;
  document.getElementById('send-btn').disabled=false;
}

function addToCart(id) {
  var item = cfg.menu.find(function(i){return i.id===id});
  if(!item) return;
  var ex = cart.find(function(c){return c.id===id});
  if(ex) ex.qty++; else cart.push(Object.assign({},item,{qty:1}));
  updateBadge();
  showToast('✓ Ajoute !');
}

function updateBadge() {
  var total = cart.reduce(function(s,i){return s+i.qty},0);
  var b = document.getElementById('cart-badge');
  b.textContent=total; b.style.display=total>0?'inline-flex':'none';
}

}

function addDish() {
  var nameFr = (document.getElementById('d-nameFr')||{}).value||'';
  var nameEn = (document.getElementById('d-nameEn')||{}).value||'';
  var price = (document.getElementById('d-price')||{}).value||'';
  var emoji = (document.getElementById('d-emoji')||{}).value||'🍽️';
  var cat = (document.getElementById('d-cat')||{}).value||'Menu';
  var desc = (document.getElementById('d-desc')||{}).value||'';
  if(!nameFr.trim()||!price) { showToast(cfg.lang==='fr'?'Nom et prix requis':'Name and price required'); return; }
  if(!cfg.menu) cfg.menu=[];
  cfg.menu.push({id:'d'+Date.now(),nameFr:nameFr.trim(),nameEn:nameEn||nameFr,price:Number(price),emoji:emoji.trim()||'🍽️',category:cat.trim()||'Menu',desc:desc});
  try{sessionStorage.setItem('bm_cfg',JSON.stringify(cfg));}catch(e){}
  showToast(cfg.lang==='fr'?'✓ Plat ajoute !':'✓ Dish added!');
  renderSettings(); renderMenu();
}

function delDish(i) {
  cfg.menu.splice(i,1);
  try{sessionStorage.setItem('bm_cfg',JSON.stringify(cfg));}catch(e){}
  showToast(cfg.lang==='fr'?'✓ Plat supprime':'✓ Dish removed');
  renderSettings(); renderMenu();
}

function changeEmoji(el,e) {
  cfg.emoji=e;
  document.querySelectorAll('#s-emojis .emoji-opt').forEach(function(x){x.classList.remove('on')});
  el.classList.add('on');
  applyConfig();
}

function changeColor(c,el) {
  if(cfg.plan!=='pro') { openUpgrade(); return; }
  cfg.color=c;
  document.querySelectorAll('.color-row .color-swatch').forEach(function(x){x.classList.remove('on')});
  el.classList.add('on');
  applyConfig();
}

function saveSettings() {
  var n=document.getElementById('s-name'); if(n) cfg.name=n.value||cfg.name;
  var d=document.getElementById('s-desc'); if(d) cfg.desc=d.value;
  var ci=document.getElementById('s-city'); if(ci) cfg.city=ci.value;
  var ty=document.getElementById('s-type'); if(ty) cfg.type=ty.value;
  applyConfig();
  try{sessionStorage.setItem('bm_cfg',JSON.stringify(cfg));}catch(e){}
  showToast(cfg.lang==='fr'?'✓ Sauvegarde !':'✓ Saved!');
}

// ============ CLIENT LINK ============
function sharClientLink() {
  if(!cfg.menu||cfg.menu.length===0) { showToast(cfg.lang==='fr'?'Ajoutez des plats d abord':'Add dishes first'); return; }
  var data = {
    name:cfg.name, type:cfg.type, city:cfg.city,
    emoji:cfg.emoji, color:cfg.color, lang:cfg.lang,
    menu:cfg.menu, apiKey:API_KEY
  };
  try {
    var encoded = btoa(encodeURIComponent(JSON.stringify(data)));
    var base = window.location.href.replace('bookmaker-app.html','');
    var url = base + 'client.html?data=' + encoded;
    if(navigator.share) {
      navigator.share({title:cfg.name+' — Commander',text:'Commandez chez '+cfg.name,url:url});
    } else if(navigator.clipboard) {
      navigator.clipboard.writeText(url).then(function(){ showToast('🔗 Lien copie !'); });
    } else {
      showToast('Lien : ' + url.substring(0,40) + '...');
    }
  } catch(e) { showToast('Erreur lors de la generation du lien'); }
}

function showToast(msg) {
  var t=document.getElementById('toast');
  t.textContent=msg; t.classList.add('show');
  setTimeout(function(){t.classList.remove('show');},2500);
}
